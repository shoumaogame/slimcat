            const hint=$('#loginHint'), name=$('#userNameSpan'), dot=$('#roleDot'), av=$('#userAvatar'), pub=$('#tabPublish');
            if(currentPlayer) { hint.style.display='none'; name.style.display=''; name.textContent=currentPlayer.nickname; dot.style.display=''; av.textContent=currentPlayer.nickname[0].toUpperCase();
                if(currentPlayer.role==='admin'){ dot.className='role-dot admin-dot'; av.className='user-avatar admin-av'; pub.style.display=''; } else { dot.className='role-dot user-dot'; av.className='user-avatar'; pub.style.display='none'; } }
            else { hint.style.display=''; name.style.display='none'; dot.style.display='none'; av.textContent='?'; av.className='user-avatar'; pub.style.display='none'; }
        }
        function updateStats() {
            $('#sTotal').textContent = orders.filter(o=>o.status!=='cancelled').length;
            $('#sPending').textContent = orders.filter(o=>o.status==='pending').length;
            $('#sDoing').textContent = orders.filter(o=>o.status==='accepted'||o.status==='doing').length;
            $('#sDone').textContent = orders.filter(o=>o.status==='done').length;
            if(currentPlayer){ $('#mPub').textContent = orders.filter(o=>o.publisher===currentPlayer.nickname&&o.status!=='cancelled').length; $('#mAcc').textContent = orders.filter(o=>o.acceptor===currentPlayer.nickname).length; }
            else { $('#mPub').textContent='-'; $('#mAcc').textContent='-'; }
        }

        function renderView() {
            document.querySelectorAll('.nav-tab').forEach(t=>t.classList.remove('active'));
            document.querySelector(`.nav-tab[data-view="${currentView}"]`)?.classList.add('active');
            if(currentView==='hall') renderHall(); else if(currentView==='my') renderMy(); else if(currentView==='publish') openPublish();
        }
        function renderHall() {
            let list = [...orders].filter(o=>o.status!=='cancelled');
            if(currentFilter==='pending') list=list.filter(o=>o.status==='pending'); else if(currentFilter==='active') list=list.filter(o=>o.status==='accepted'||o.status==='doing'); else if(currentFilter==='done') list=list.filter(o=>o.status==='done');
            list.sort((a,b)=>({pending:0,accepted:1,doing:2,done:3}[a.status]??9)-({pending:0,accepted:1,doing:2,done:3}[b.status]??9)||new Date(b.created_at)-new Date(a.created_at));
            let html = `<div class="filters"><button class="filter-chip ${currentFilter==='all'?'active':''}" onclick="setFilter('all')">全部</button><button class="filter-chip ${currentFilter==='pending'?'active':''}" onclick="setFilter('pending')">待接单</button><button class="filter-chip ${currentFilter==='active'?'active':''}" onclick="setFilter('active')">进行中</button><button class="filter-chip ${currentFilter==='done'?'active':''}" onclick="setFilter('done')">已完成</button></div><div class="order-grid">`;
            if(!list.length) html+='<div class="empty-state"><div class="empty-icon">📭</div><p>暂无订单</p></div>';
            else list.forEach(o=> html+=orderCard(o));
            html+='</div>'; $('#mainContent').innerHTML=html;
            document.querySelectorAll('.order-card').forEach(card=>{ card.addEventListener('click',function(e){ if(e.target.closest('button')) return; const o=orders.find(x=>x.id===this.dataset.id); if(o?.status==='pending'&&currentPlayer?.role!=='admin'){ if(confirm('确认接单？')) acceptOrder(o.id); } }); });
        }
        function orderCard(o) {
            const cls={pending:'status-pending',accepted:'status-accepted',doing:'status-doing',done:'status-done'}[o.status];
            const tcls={rank:'badge-rank',peak:'badge-peak',hero:'badge-hero',play:'badge-play'}[o.type]||'';
            let btn='';
            if(o.status==='pending'&&currentPlayer?.role!=='admin'&&currentPlayer) btn=`<button class="btn btn-accent btn-sm" onclick="event.stopPropagation();acceptOrder('${o.id}')">抢单</button>`;
            else if(o.status==='accepted'&&o.acceptor===currentPlayer?.nickname) btn=`<button class="btn btn-primary btn-sm" onclick="event.stopPropagation();startOrder('${o.id}')">开始打</button>`;
            else if(o.status==='doing'&&o.acceptor===currentPlayer?.nickname) btn=`<button class="btn btn-success btn-sm" onclick="event.stopPropagation();completeOrder('${o.id}')">完成</button>`;
            else if(o.status==='pending'&&currentPlayer?.role==='admin') btn=`<button class="btn btn-danger btn-sm" onclick="event.stopPropagation();cancelOrder('${o.id}')">取消</button>`;
            return `<div class="order-card" data-id="${o.id}"><div class="card-top"><span class="badge-type ${tcls}">${o.type_label||''}</span><span class="badge-status ${cls}">${statusLabel(o.status)}</span></div><div class="card-title">${o.target_rank||''}</div><div class="card-meta"><span>📍 ${o.position||'不限'}</span><span>📋 ${o.id}</span>${o.acceptor?`<span>👤 ${o.acceptor}</span>`:''}</div>${o.note?`<div class="card-meta">💬 ${o.note}</div>`:''}<div class="card-price">¥ ${o.price}</div><div class="card-bottom"><span class="card-time">${timeAgo(o.created_at)}</span>${btn}</div></div>`;
        }
        function renderMy() {
            if(!currentPlayer){ $('#mainContent').innerHTML='<div class="empty-state"><p>请先登录</p></div>'; return; }
            const my=orders.filter(o=>o.publisher===currentPlayer.nickname||o.acceptor===currentPlayer.nickname);
            my.sort((a,b)=>new Date(b.updated_at||b.created_at)-new Date(a.updated_at||a.created_at));
            let html='<div class="order-grid">'; if(!my.length) html+='<div class="empty-state">暂无相关订单</div>'; else my.forEach(o=>html+=orderCard(o)); html+='</div>'; $('#mainContent').innerHTML=html;
        }
        window.setFilter = f => { currentFilter=f; renderHall(); };
        window.openPublish = () => { if(!currentPlayer||currentPlayer.role!=='admin') return; $('#publishModal').style.display='flex'; };
        window.closePublishModal = () => $('#publishModal').style.display='none';
        window.handlePublish = async (e) => {
            e.preventDefault();
            const d={type:$('#pubType').value, targetRank:$('#pubRank').value, position:$('#pubPos').value, price:$('#pubPrice').value, note:$('#pubNote').value};
            if(!d.price||d.price<1) return toast('请输入价格','warn');
            await publishOrder(d); closePublishModal(); currentView='hall'; renderView();
        };
        window.acceptOrder = acceptOrder; window.startOrder = startOrder; window.completeOrder = completeOrder; window.cancelOrder = cancelOrder;

        document.getElementById('navTabs').addEventListener('click',e=>{ const tab=e.target.closest('.nav-tab'); if(!tab) return; const v=tab.dataset.view; if(v==='publish'){ openPublish(); return; } currentView=v; currentFilter='all'; renderView(); });
        document.getElementById('publishModal').addEventListener('click',e=>{ if(e.target===e.currentTarget) closePublishModal(); });
        document.getElementById('authModal').addEventListener('click',e=>{ if(e.target===e.currentTarget) closeAuthModal(); });
        document.addEventListener('keydown',e=>{ if(e.key==='Escape'){ closePublishModal(); closeAuthModal(); } });

        (async () => {
            if(!initSB()) return;
            const saved = localStorage.getItem('sc_player');
            if(saved) try { currentPlayer = JSON.parse(saved); } catch { currentPlayer=null; }
            updateUI(); await loadOrders(); subRT(); updateStats();
        })();
